## Resolucion Ejercicio 1

```scala
package semana9.tallerB2S9

import rx.lang.scala.Observable
import scala.concurrent.duration._
import scala.util.Random

object Ejercicio1 extends App {

  val sensor1: Observable[Int] = Observable.interval(1.second).map(_ => Random.between(20, 31))
  val sensor2: Observable[Int] = Observable.interval(1.5.second).map(_ => Random.between(30, 41))

  val sensorStream = sensor1.zip(sensor2)

  sensorStream
    .take(10)
    .subscribe(
      onNext = { case (s1, s2) =>
        val diferencia = Math.abs(s1 - s2)

        println(s"Sensor 1: $s1¬∞C, Sensor 2: $s2¬∞C, Diferencia: $diferencia¬∞C")

        if (diferencia > 5) {
          println(s"ALERT: Significant difference detected: ${diferencia}¬∞C")
        }
      },
      onError = e => println(s"Error: ${e.getMessage}"),
      onCompleted = () => println("Sensor stopped")
    )

  Thread.sleep(16000)
}
```

package semana9.tallerB2S9

import rx.lang.scala.Observable
import scala.concurrent.duration._
import scala.util.Random

object Ejercicio2 extends App {

  val sensor1: Observable[Int] = Observable.interval(1.second).map(_ => Random.between(20, 31))
  val sensor2: Observable[Int] = Observable.interval(1.second).map(_ => Random.between(40, 61))
  val sensor3: Observable[Int] = Observable.interval(1.second).map(_ => Random.between(950, 1051))

  val combinado = sensor1.zip(sensor2)

  val sensorStream = combinado.zip(sensor3)

  sensorStream
    .take(20)
    .subscribe(
      onNext = {
        case ((temp, hum), presion) =>

          var alertMessage = ""

          if (temp < 20 || temp > 30) {
              alertMessage += s"Temperatura ${temp}¬∞C fuera de rango. "
          }
          if (hum < 40 || hum > 60) {
              alertMessage += s"Humedad ${hum}% fuera de rango. "
          }
          if (presion < 950 || presion > 1050) {
              alertMessage += s"Presi√≥n ${presion} fuera de rango. "
          }

          println(s"Temperature: ${temp}¬∞C, Humidity: ${hum}%, Pressure: ${presion} hPa")

          if (alertMessage.nonEmpty) {
              println(s"ALERT: Medidas fuera de rango: ${alertMessage}")
          }
      },
      onError = e => println(s" Error: ${e.getMessage}"),
      onCompleted = () => println("Sensor stopped")
    )

  Thread.sleep(21000)
}
